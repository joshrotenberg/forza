Configuration Examples

Ready-to-use configuration examples. Copy the one closest to your needs and adjust the repo name.

Minimal

The simplest possible setup — one repo, one bug route, all defaults. Use this as your starting point.

# forza minimal configuration
#
# The simplest possible forza setup. One repo, one bug route, all defaults.
# Use this as your starting point — copy, change the repo name, and go.
#
# When to use: you want to try forza with zero ceremony.

[global]
model = "claude-sonnet-4-6"

[security]
authorization_level = "contributor"

[repos."your-org/your-repo"]

[repos."your-org/your-repo".routes.bugfix]
type = "issue"
label = "bug"
workflow = "bug"

Rust project

Standard setup for a Rust codebase: fmt/clippy/test validation, three issue routes (bugs, features, chores), auto-fix condition route.

# forza configuration — typical Rust project
#
# Standard setup for a Rust codebase: fmt/clippy/test validation, three issue
# routes (bugs, features, chores), auto-merge enabled for approved green PRs.
#
# When to use: you maintain a Rust library or CLI and want forza to handle the
# full issue-to-PR lifecycle with Rust-specific quality checks.

[global]
model = "claude-sonnet-4-6"
gate_label = "forza:ready"
branch_pattern = "automation/{issue}-{slug}"
auto_merge = false   # set true to merge PRs automatically once CI is green

# Cost guardrails.
max_cost_per_issue = 5.00
max_cost_per_hour = 20.00

# ── Security ─────────────────────────────────────────────────────────

[security]
# contributor: processes issues from repo contributors and above.
authorization_level = "contributor"

# ── Validation ───────────────────────────────────────────────────────
#
# These commands run between stages. All must pass before the next stage starts.

[validation]
commands = [
    "cargo fmt --all -- --check",
    "cargo clippy --all-targets --all-features -- -D warnings",
    "cargo test --lib --all-features",
]

# ── Agent config ─────────────────────────────────────────────────────

[agent_config]
# Inject a Rust-specific skill file into every agent invocation.
skills = ["./skills/rust.md"]

# ── Routes ───────────────────────────────────────────────────────────

[repos."your-org/your-rust-crate"]

[repos."your-org/your-rust-crate".routes.bugfix]
type = "issue"
label = "bug"
workflow = "bug"
concurrency = 1        # one bug at a time to avoid merge conflicts
poll_interval = 60     # check every minute — bugs are urgent
model = "claude-opus-4-6"  # use a more capable model for bug investigation

[repos."your-org/your-rust-crate".routes.features]
type = "issue"
label = "enhancement"
workflow = "feature"
concurrency = 2
poll_interval = 300    # check every 5 minutes

[repos."your-org/your-rust-crate".routes.chores]
type = "issue"
label = "chore"
workflow = "chore"
concurrency = 3
poll_interval = 600    # check every 10 minutes

# Automatically fix CI failures and conflicts on forza-created PRs.
[repos."your-org/your-rust-crate".routes.auto-fix]
type = "pr"
condition = "ci_failing_or_conflicts"
workflow = "pr-fix"
scope = "forza_owned"
concurrency = 2
poll_interval = 60
max_retries = 3        # apply forza:needs-human after 3 failed attempts

# ── Per-stage hooks ──────────────────────────────────────────────────
#
# Format code after every implement stage, run tests before opening a PR.

[stage_hooks.implement]
post = ["cargo fmt --all"]

[stage_hooks.open_pr]
pre = ["cargo test --all-features"]

PR maintenance

Focused on keeping open PRs healthy. Condition routes fire automatically based on PR state — no label needed.

# forza configuration — PR maintenance / reactive workflows
#
# Focused on keeping open PRs healthy. Condition routes fire automatically
# based on PR state — no label needed. forza watches for CI failures,
# merge conflicts, and approved-and-green PRs, then acts accordingly.
#
# When to use: you want forza to act as a PR janitor: fix broken CI, resolve
# conflicts, and merge approved PRs without manual intervention.

[global]
model = "claude-sonnet-4-6"
# No gate_label — condition routes bypass the label gate by default.
max_concurrency = 4

max_cost_per_issue = 3.00
max_cost_per_hour = 15.00

# ── Security ─────────────────────────────────────────────────────────

[security]
# contributor: only act on PRs from known contributors.
authorization_level = "contributor"

# ── Validation ───────────────────────────────────────────────────────

[validation]
commands = [
    "cargo fmt --all -- --check",
    "cargo clippy --all-targets -- -D warnings",
    "cargo test --lib",
]

# ── Routes ───────────────────────────────────────────────────────────
#
# All routes here are condition-based. They fire automatically when the
# described PR state is detected — no label application required.
#
# scope = "forza_owned" (the default) restricts evaluation to PRs on
# branches that forza created. Set scope = "all" to watch every open PR.

[repos."your-org/your-repo"]

# Fix CI failures: rerun, then attempt a targeted code fix if still failing.
[repos."your-org/your-repo".routes.fix-ci]
type = "pr"
condition = "ci_failing"
workflow = "pr-fix"
scope = "forza_owned"
concurrency = 2
poll_interval = 60     # check every minute — CI failures block progress
max_retries = 3        # give up and apply forza:needs-human after 3 attempts
model = "claude-opus-4-6"   # CI fixes can be complex; use the capable model

# Resolve merge conflicts automatically.
[repos."your-org/your-repo".routes.fix-conflicts]
type = "pr"
condition = "has_conflicts"
workflow = "pr-fix"
scope = "forza_owned"
concurrency = 2
poll_interval = 120
max_retries = 2

# Handle both CI failures and conflicts in one route (alternative to the two above).
# Uncomment this and remove the two routes above if you prefer a single route.
# [repos."your-org/your-repo".routes.fix-all]
# type = "pr"
# condition = "ci_failing_or_conflicts"
# workflow = "pr-fix"
# scope = "forza_owned"
# concurrency = 2
# poll_interval = 60
# max_retries = 3

# Merge PRs that are approved and have passing CI.
[repos."your-org/your-repo".routes.auto-merge]
type = "pr"
condition = "approved_and_green"
workflow = "merge"
scope = "forza_owned"
concurrency = 1        # serialize merges to avoid race conditions
poll_interval = 120

    { kind = "merge" },

Security-strict

A locked-down configuration for high-stakes or public-facing repos. Sandbox authorization, no auto-merge, explicit author allowlist, and a required approval label.

# forza configuration — security-strict setup
#
# A locked-down configuration for high-stakes or public-facing repos.
# Sandbox authorization, no auto-merge, explicit author allowlist, and a
# required approval label before any work begins. Every PR requires a human
# review before it can be merged.
#
# When to use: production repos, security-sensitive codebases, or any project
# where you want maximum human oversight of automated changes.

[global]
model = "claude-sonnet-4-6"
gate_label = "forza:ready"    # explicit opt-in required on every issue/PR
branch_pattern = "automation/{issue}-{slug}"
auto_merge = false            # never merge automatically — human must merge

max_cost_per_issue = 2.00    # tighter cost cap per issue
max_cost_per_hour = 10.00

# ── Security ─────────────────────────────────────────────────────────

[security]
# sandbox: the most restrictive level. The agent may only read files and
# suggest changes; it cannot execute arbitrary commands or write to disk
# outside of the controlled workflow.
authorization_level = "sandbox"

# Only process issues created by explicitly listed authors.
allowed_authors = ["your-github-username", "trusted-collaborator"]

# An additional label that must be present on the issue before processing.
# Unlike gate_label (which triggers the run and is removed), require_label
# is a permanent security gate that is never removed automatically.
# A human must apply this label to explicitly approve processing.
require_label = "security:approved"

# ── Validation ───────────────────────────────────────────────────────
#
# Run a comprehensive check suite between every stage.

[validation]
commands = [
    "cargo fmt --all -- --check",
    "cargo clippy --all-targets --all-features -- -D warnings",
    "cargo test --lib --all-features",
    "cargo test --test '*' --all-features",
    "cargo audit",   # check for known vulnerabilities in dependencies
]

# ── Agent config ─────────────────────────────────────────────────────

[agent_config]
# Inject security-focused instructions into every agent invocation.
skills = ["./skills/security.md"]
append_system_prompt = "You are operating in a high-security context. Do not introduce new dependencies without explicit justification. Prefer conservative, minimal changes."

# ── Routes ───────────────────────────────────────────────────────────
#
# Only two routes: bugs and the triage/research workflow. No feature or chore
# routes — new feature work requires a human to design and guide.

[repos."your-org/sensitive-repo"]

[repos."your-org/sensitive-repo".routes.bugfix]
type = "issue"
label = "bug"
workflow = "bug"
concurrency = 1        # one at a time — avoid overlapping changes
poll_interval = 120
max_retries = 1        # escalate to forza:needs-human after a single failure
model = "claude-opus-4-6"   # use the most capable model for bug analysis

[repos."your-org/sensitive-repo".routes.research]
type = "issue"
label = "research"
workflow = "research"
concurrency = 1
poll_interval = 300

# ── Per-stage hooks ──────────────────────────────────────────────────
#
# Run the full test suite before opening any PR.

[stage_hooks.open_pr]
pre = [
    "cargo test --all-features",
    "cargo audit",
]

Full runner

Complete example showing all available options: multi-repo, cost limits, custom workflow templates, agent config, and per-stage hooks.

# forza runner configuration
#
# The runner watches GitHub for labeled issues and PRs, then processes them
# through configurable workflow templates. Each route is a named lane with
# its own label trigger, workflow, concurrency, and polling frequency.
#
# Multi-repo mode: define one [repos."owner/name"] table per repository.
# Legacy single-repo mode: set repo = "owner/name" under [global] and define
# [routes] at the top level.

# ── Global settings ──────────────────────────────────────────────────
#
# These settings apply to all repos.

[global]
# Maximum concurrent runs across all routes and repos.
max_concurrency = 5

# Cost guardrails.
max_cost_per_issue = 5.00     # fail if a single issue exceeds this
max_cost_per_hour = 20.00     # pause all routes if hourly spend exceeds this

# Default agent and model (overridable per route).
agent = "claude"
model = "claude-sonnet-4-6"

# Gate label — only process issues/PRs with this label (applied globally).
# Provides explicit opt-in. Remove to process all matching routes.
gate_label = "runner:ready"

# Default branch pattern. {issue} = number, {slug} = slugified title.
branch_pattern = "automation/{issue}-{slug}"

# Automatically merge PRs once all CI checks pass. Default: false.
# auto_merge = false

# ── Security ─────────────────────────────────────────────────────────

[security]
# Authorization level: sandbox, local, contributor, trusted
authorization_level = "contributor"

# Only process issues from these authors. Empty = authenticated user only.
allowed_authors = []

# Additional label required on an issue before processing begins.
# Unlike gate_label (the opt-in trigger that gets removed at lease time),
# require_label is a security gate checked after the issue is fetched and
# never removed automatically. Useful for enforcing a human-review stamp.
# require_label = "security:approved"

# ── Validation ───────────────────────────────────────────────────────

# Commands run between stages (in addition to per-stage hooks).
[validation]
commands = [
    "cargo fmt --all -- --check",
    "cargo clippy --all-targets -- -D warnings",
    "cargo test --lib",
]

# ── Repos ────────────────────────────────────────────────────────────
#
# Define one [repos."owner/name"] section per repository to watch.
# Each repo has an optional local checkout path and its own route table.
#
# repo_dir — path to the local git checkout. Defaults to current directory.
# routes   — named routes specific to this repo (same structure as legacy
#            top-level [routes]).

[repos."joshrotenberg/claude-wrapper"]
# repo_dir = "/path/to/claude-wrapper"   # optional local checkout path

[repos."joshrotenberg/claude-wrapper".routes.bugfix]
type = "issue"
label = "bug"
workflow = "bug"
concurrency = 1           # one bug at a time (avoid conflicts)
poll_interval = 60        # check every minute (bugs are urgent)
# model = "claude-opus-4-6"  # override model for complex bugs

[repos."joshrotenberg/claude-wrapper".routes.features]
type = "issue"
label = "enhancement"
workflow = "feature"
concurrency = 2           # two features in parallel
poll_interval = 300       # check every 5 minutes

[repos."joshrotenberg/claude-wrapper".routes.chores]
type = "issue"
label = "chore"
workflow = "chore"
concurrency = 3
poll_interval = 600

[repos."joshrotenberg/other-project"]
# repo_dir = "/path/to/other-project"

[repos."joshrotenberg/other-project".routes.bugfix]
type = "issue"
label = "bug"
workflow = "bug"
concurrency = 1
poll_interval = 60

[repos."joshrotenberg/other-project".routes.research]
type = "issue"
label = "research"
workflow = "research"
concurrency = 3
poll_interval = 3600      # check hourly (not urgent)

# ── Condition routes ──────────────────────────────────────────────────────────
#
# Condition routes fire automatically based on PR state rather than a label.
# Supported conditions:
#   ci_failing            — CI checks are failing on the PR
#   has_conflicts         — the PR has merge conflicts
#   ci_failing_or_conflicts — CI failing or conflicts (most common)
#   approved_and_green    — PR is approved and CI is passing (auto-merge trigger)
#
# scope controls which PRs are evaluated:
#   forza_owned (default) — only PRs on branches created by forza
#   all                   — all open PRs in the repo
#
# No gate_label needed — condition routes bypass the label gate.

[repos."joshrotenberg/claude-wrapper".routes.auto-fix]
type = "pr"
condition = "ci_failing_or_conflicts"  # fire when CI fails or conflicts detected
workflow = "pr-fix"
scope = "forza_owned"         # only touch forza-created branches
concurrency = 2
poll_interval = 60            # check every minute
max_retries = 3               # apply forza:needs-human after 3 failures
# model = "claude-opus-4-6"   # use a more capable model for tricky fixes

# ── Workflow templates ───────────────────────────────────────────────
#
# Built-in templates: bug, feature, chore, research
# Define custom templates here. Custom overrides built-in with same name.

[[workflow_templates]]
name = "hotfix"
stages = [
    { kind = "implement", max_retries = 1 },
    { kind = "test", max_retries = 2 },
    { kind = "open_pr" },
]

[[workflow_templates]]
name = "discuss"
stages = [
    { kind = "research" },
    { kind = "comment" },
]

# ── Agent config ─────────────────────────────────────────────────────

# [agent_config]
# skills = ["./skills/rust-workspace.md"]
# mcp_config = ".mcp.json"
# append_system_prompt = "Project-specific rules."

# ── Per-stage hooks ──────────────────────────────────────────────────

[stage_hooks.implement]
post = ["cargo fmt --all"]

[stage_hooks.test]
post = ["cargo test --lib"]

[stage_hooks.open_pr]
pre = ["cargo test --all-features"]
